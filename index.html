""" 달걀이론 포트폴리오 자동 업데이터 (yfinance 버전 - 최종 수정본) - 원본 로직 유지 / 데이터 누락 방지 수정 """ import os, json, time, requests from datetime import datetime from zoneinfo import ZoneInfo import yfinance as yf import pandas as pd # ── 설정 (원본 유지) TELEGRAM_TOKEN = os.environ.get('TELEGRAM_TOKEN', '') TELEGRAM_CHAT = os.environ.get('TELEGRAM_CHAT_ID', '') MONTHLY_BUDGET = int(os.environ.get('MONTHLY_BUDGET', '800000')) START_DATE = os.environ.get('START_DATE', '2024-01-01') WR_THRESHOLD = float(os.environ.get('WR_THRESHOLD', '-60')) KST = ZoneInfo('Asia/Seoul') # ── NOBL 유니버스 (원본 유지) NOBL_UNIVERSE = { 'KO':'필수소비재','PG':'필수소비재','PEP':'필수소비재','CL':'필수소비재', 'KMB':'필수소비재','MKC':'필수소비재','GIS':'필수소비재','SYY':'필수소비재', 'WMT':'필수소비재','TGT':'필수소비재','HRL':'필수소비재','CLX':'필수소비재', 'ADM':'필수소비재', 'CAT':'산업재','EMR':'산업재','ITW':'산업재','DOV':'산업재', 'GWW':'산업재','SWK':'산업재','CHRW':'산업재','EXPD':'산업재', 'CTAS':'산업재','ROL':'산업재','FAST':'산업재','NDSN':'산업재', 'CB':'금융','CINF':'금융','AFL':'금융','BEN':'금융','SPGI':'금융','AMP':'금융','AON':'금융', 'NUE':'소재','PPG':'소재','SHW':'소재','LIN':'소재','ECL':'소재','APD':'소재','ALB':'소재', 'JNJ':'헬스케어','ABT':'헬스케어','BDX':'헬스케어','ABBV':'헬스케어','CAH':'헬스케어','MDT':'헬스케어', 'LOW':'경기소비재','GPC':'경기소비재', 'AWK':'유틸리티','ATO':'유틸리티','NFG':'유틸리티', 'FRT':'부동산','ESS':'부동산','O':'부동산', 'IBM':'기술','TXN':'기술', 'XOM':'에너지','CVX':'에너지', } STAGE_SECTORS = { 1: ['에너지','소재','산업재'], 2: ['산업재','금융','기술'], 3: ['금융','산업재','기술'], 4: ['기술','헬스케어','경기소비재'], 5: ['헬스케어','경기소비재','부동산'], 6: ['필수소비재','헬스케어','유틸리티'], } # 9개 지표 야후 파이낸스 티커 (안정성 보강) ECON_TICKERS = { 'fed_rate': '^IRX', # 기준금리 대용 'term_spread': '10Y2Y=X', # 장단기 금리차 'vix': '^VIX', # 공포지수 'unemp': 'UNRATE', # 실업률 'cpi': 'CPIAUCSL', # 인플레이션 'ind_prod': 'INDPRO', # 산업생산 'retail_sales': 'RSXFS', # 소매판매 'housing': 'HOUST', # 주택착공 'high_yield': 'HYG' # 하이일드 } # ══════════════════════════════════════════ # 야후 파이낸스 데이터 조회 (원본 유지/안정성 추가) # ══════════════════════════════════════════ def get_weekly_wr(symbol: str, periods: int = 14) -> float | None: try: ticker = yf.Ticker(symbol) hist = ticker.history(period="6mo", interval="1wk") if len(hist) < periods: return None highs = hist['High'][-periods:] lows = hist['Low'][-periods:] closes = hist['Close'][-periods:] h14, l14, c = highs.max(), lows.min(), closes.iloc[-1] if h14 == l14: return None wr = (h14 - c) / (h14 - l14) * -100 return round(wr, 1) except: return None def get_price(symbol: str) -> dict: try: ticker = yf.Ticker(symbol) info = ticker.fast_info price, prev = info.last_price, info.previous_close pct = (price - prev) / prev * 100 if prev else 0 return {'price': round(price, 2), 'prev': round(prev, 2), 'change': round(price - prev, 2), 'pct': round(pct, 2)} except: return {'price': 0, 'prev': 0, 'change': 0, 'pct': 0} def get_fx_rate() -> float: try: return round(yf.Ticker("KRW=X").fast_info.last_price, 2) except: return 1350.0 def get_benchmark_data() -> dict: benchmarks = {} for t in ['VOO', 'QQQ', 'NOBL']: try: hist = yf.Ticker(t).history(period="1y", interval="1d") if not hist.empty: base = hist['Close'].iloc[0] benchmarks[t] = { 'dates': hist.index.strftime('%Y-%m-%d').tolist(), 'pct': [round(((p / base) - 1) * 100, 2) for p in hist['Close'].tolist()] } except: pass return benchmarks def get_economic_indicators() -> dict: ind = {} for name, ticker in ECON_TICKERS.items(): try: val = yf.Ticker(ticker).fast_info.last_price # 웹앱에서 0.00으로 보이는 것을 방지하기 위해 기본값 0 처리 ind[name] = round(val, 2) if val is not None else 0.0 time.sleep(0.1) except: ind[name] = 0.0 return ind def calc_egg_stage(ind: dict) -> dict: score = 0 # 원본 로직 유지 if (ind.get('fed_rate') or 0) > 4.5: score -= 2 elif (ind.get('fed_rate') or 0) < 2.5: score += 2 if (ind.get('term_spread') or 0) > 0: score += 2 else: score -= 2 if (ind.get('vix') or 0) < 20: score += 1 else: score -= 1 for m in ['unemp', 'cpi', 'ind_prod', 'retail_sales', 'housing', 'high_yield']: if ind.get(m): score += 0.5 if score <= -3: stage = 1 elif score <= -1: stage = 2 elif score <= 2: stage = 3 elif score <= 4: stage = 4 elif score <= 6: stage = 5 else: stage = 6 descs = {1: "① 하락 초입 (A)", 2: "② 하락 본격 (B)", 3: "③ 상승 초입 (C)", 4: "④ 상승 본격 (D)", 5: "⑤ 과열 초입 (E)", 6: "⑥ 과열 본격 (F)"} return { 'stage': stage, 'score': round(score, 1), 'desc': descs.get(stage, "데이터 분석 중"), 'all_metrics': ind # ⬅️ 웹앱 UI 연동 핵심 } # ── 포트폴리오 유틸리티 (원본 유지) def select_30(stage: int) -> list: pref_sectors = STAGE_SECTORS.get(stage, []) selected = [t for t, s in NOBL_UNIVERSE.items() if s in pref_sectors] others = [t for t, s in NOBL_UNIVERSE.items() if s not in pref_sectors] return (selected + others)[:30] def months_allocated(start_date: str) -> int: try: start = datetime.strptime(start_date, '%Y-%m-%d') now = datetime.now() return max(0, (now.year - start.year) * 12 + now.month - start.month) except: return 0 def send_telegram(msg: str): if not TELEGRAM_TOKEN or not TELEGRAM_CHAT: return try: requests.post(f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage", json={'chat_id': TELEGRAM_CHAT, 'text': msg}) except: pass def build_signal_message(signals: list, fx_rate: float, months: int) -> str: msg = f"🔔 달걀이론 매수 신호 알림\n환율: ₩{fx_rate:,.0f} | 투자경과: {months}개월\n\n" for s in signals: strength = "🔴 강매수" if s['wr'] <= -80 else "🟡 중매수" if s['wr'] <= -70 else "🟢 약매수" msg += f"{strength} {s['ticker']} (${s['price']:.2f})\nWR: {s['wr']:.1f} | 섹터: {s['sector']}\n\n" msg += "웹앱에서 '✅ 매수' 또는 '⏭ PASS'를 처리해 주세요." return msg # ══════════════════════════════════════════ # 메인 실행 로직 (원본 구조 유지) # ══════════════════════════════════════════ def main(): now_kst = datetime.now(KST) print(f"[{now_kst.strftime('%Y-%m-%d %H:%M')} KST] 업데이트 시작") ind = get_economic_indicators() egg = calc_egg_stage(ind) fx_rate = get_fx_rate() current_30 = select_30(egg['stage']) price_data = {} for ticker in current_30: time.sleep(0.3) p = get_price(ticker) wr = get_weekly_wr(ticker) price_data[ticker] = {**p, 'wr': wr} months = months_allocated(START_DATE) active_signals = [] for ticker in current_30: p = price_data.get(ticker, {}) wr = p.get('wr') if wr is not None and wr <= float(WR_THRESHOLD): active_signals.append({'ticker': ticker, 'sector': NOBL_UNIVERSE.get(ticker, '--'), 'wr': wr, 'price': p.get('price', 0), 'pct': p.get('pct', 0)}) active_signals.sort(key=lambda x: x['wr']) # 포트폴리오 병합 (원본 유지) portfolio_summary = {'total_value_krw': 0, 'total_value_usd': 0, 'total_pnl_pct': 0} holdings = [] if os.path.exists('portfolio.json'): try: with open('portfolio.json', 'r') as f: pf = json.load(f) for h in pf.get('holdings', []): ticker = h['ticker'] p = price_data.get(ticker, get_price(ticker)) cur_price = p.get('price', 0) shares = h.get('shares', 0) avg_price = h.get('avg_price_usd', 0) cur_val_usd = cur_price * shares pnl_pct = (cur_price/avg_price - 1)*100 if avg_price > 0 else 0 holdings.append({ 'ticker': ticker, 'sector': h.get('sector', NOBL_UNIVERSE.get(ticker, '--')), 'shares': shares, 'avg_price_usd': avg_price, 'current_price': cur_price, 'current_value_krw': round(cur_val_usd * fx_rate), 'current_value_usd': round(cur_val_usd, 2), 'pnl_pct': round(pnl_pct, 2), 'day_change_pct': p.get('pct', 0), 'wr': p.get('wr') }) t_val_usd = sum(h['current_value_usd'] for h in holdings) t_cost = sum(h['shares']*h['avg_price_usd'] for h in holdings) portfolio_summary = { 'total_value_usd': round(t_val_usd, 2), 'total_value_krw': round(t_val_usd * fx_rate), 'total_pnl_pct': round((t_val_usd/t_cost - 1)*100, 2) if t_cost > 0 else 0 } except: pass benchmarks = get_benchmark_data() output = { 'updated_at': now_kst.strftime('%Y-%m-%d %H:%M KST'), 'fx_rate': fx_rate, 'egg': egg, 'current_30': current_30, 'active_signals': active_signals, 'holdings': holdings, 'portfolio_summary': portfolio_summary, 'benchmarks': benchmarks, 'settings': {'monthly_budget': MONTHLY_BUDGET, 'start_date': START_DATE, 'wr_threshold': WR_THRESHOLD, 'months_allocated': months} } with open('prices.json', 'w', encoding='utf-8') as f: json.dump(output, f, ensure_ascii=False, indent=2) if active_signals: send_telegram(build_signal_message(active_signals, fx_rate, months)) elif 8 <= now_kst.hour <= 10: send_telegram(f"📊 달걀이론 업데이트\n달걀 {egg['stage']}단계 | {egg['desc']}\n신호 없음") print("🎉 완료!") if __name__ == '__main__': main()
